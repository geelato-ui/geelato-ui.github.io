# 事件

# action

事件操作，对应一个按钮或链接。

// 对应一个按钮或链接
export default {
  actions: [{
    text: '保存',
    icon: 'plus',
    show: false,
    type: 'primary',
    fn: 'save',
    // opener、content、modal，默认为content
    ctx: 'content',
    params: {},
    then: {
      fn: 'close',
      ctx: 'modal',
      params: {},
      then: {},
      fail: {}
    }
  }, {
    fn: 'close',
    text: '取消',
    ctx: 'modal'
  }, {
    fn: 'openerFnExample',
    text: '回调',
    ctx: 'opener',
    params: {info: '$ctx.$form.id'},
    then: {
      fn: 'close',
      ctx: 'modal',
      params: {},
      then: {},
      fail: {}
    }
  }]
}

# on

事件侦听，响应一个$emit事件。

// 侦听来之content的事件selectItem，并执行打开组件的方法openerFnExampleByContent，最后执行modal的关闭操作。
export default {
  on: [{
    fn: 'selectItem',
    ctx: 'content',
    then: {
      fn: 'openerFnExampleByContent',
      ctx: 'opener',
      then: {
        fn: 'close',
        ctx: 'modal'
      }
    }
  }]
}
