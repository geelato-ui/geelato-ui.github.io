import formData from '../../../Form/Base/UserFormData.js'
import tableData from '../../../Table/Base/data.js'
import treeData from '../../../Tree/Base/data.js'


export default {
  cardMap: {
    table1: {
      id: 'table1',
      title: '这是一个列表Card',
      actions: [{title: '更多'}],
      bordered: true,
      style: '',
      type: 'GlTable',

      // 权限描述符
      permission: 'table1'

      id: 'form1',
      title: '这是一个表单Card',
      type: 'GlForm',
      opts: {opts: formData}
